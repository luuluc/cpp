#include <bits/stdc++.h>
	using namespace std;
	int n , ok ;
    int a[1000];
	void kt(){
        for( int i = 1  ; i<= n ;i++ )
            a[i] = 0 ;
        
    }
	void sinh(){
        int i = n ; 
        while( i> 0 && a[i] == 1 ){
            a[i] = 0 ;
            i--;
        }
        if( i == 0 ) ok = 0;
            else a[i] = 1 ;
    }
	int  main(){
        int t;
        cin>> t;
        while(t--){
            cin>> n;
		ok = 1 ;
            kt();
            while(ok == 1){
            for(int  i =1 ;i<= n ; i++  ) cout<< a[i]<<" ";
            cout<< endl;
                sinh();
            }
        }
    }
